$$
	\pi_{k,n}=\sum^n_{i=k}i \qquad (x \space primo \space i)
$$

$$
	primo (x)= 
	\begin{cases}
		1 & \text{se x è primo} \\
		0 & \text{altrimenti} \\
	\end{cases}
$$
